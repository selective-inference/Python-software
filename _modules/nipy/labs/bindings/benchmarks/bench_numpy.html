[docs]def time_ratio(t0,t1):
    if t1==0:
        return np.inf
    else:
        return t0/t1



[docs]def time_copy_vector(x):
    t0 = time.clock()
    y0 = copy_vector(x, 0) 
    dt0 = time.clock()-t0
    t1 = time.clock()
    y1 = copy_vector(x, 1) 
    dt1 = time.clock()-t1
    ratio = time_ratio(dt0,dt1)
    print('  using fff_array: %f sec' % dt0)
    print('  using numpy C API: %f sec' % dt1)
    print('  ratio: %f' % ratio)
